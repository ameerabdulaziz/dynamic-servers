Quick Examples 

---
- name: Update server packages
  hosts: servers
  become: yes
  tasks:
    - name: Update package cache
      apt:
        update_cache: yes
      when: ansible_os_family == "Debian"
    
    - name: Upgrade all packages
      apt:
        upgrade: dist
      when: ansible_os_family == "Debian"

---
- name: Deploy web application
  hosts: servers
  become: yes
  vars:
    app_name: "{{ app_name | default('my-app') }}"
    app_port: "{{ app_port | default(3000) }}"
  tasks:
    - name: Install Node.js
      apt:
        name: nodejs
        state: present
    
    - name: Install npm
      apt:
        name: npm
        state: present
    
    - name: Create app directory
      file:
        path: "/opt/{{ app_name }}"
        state: directory
        owner: www-data
        group: www-data
